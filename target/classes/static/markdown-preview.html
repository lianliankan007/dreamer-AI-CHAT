    console.log(message);
    return {
        user: name,
        greeting: message,
        timestamp: new Date().toISOString()
    };
}

// 调用函数

# Python代码块示例
def calculate_fibonacci(n):
    """计算斐波那契数列"""
    if n <= 1:
        return n
    else:
        return calculate_fibonacci(n-1) + calculate_fibonacci(n-2)

# 生成前10个斐波那契数
fibonacci_sequence = [calculate_fibonacci(i) for i in range(10)]
print(f"斐波那契数列: {fibonacci_sequence}")

🚀 Dreamer AI Chat 功能介绍

Dreamer AI Chat是一个现代化的智能对话系统，支持多种AI模型和丰富的Markdown渲染功能。

设计理念：简洁、优雅、高效

核心特性

  1. 多模型支持：集成多个主流AI模型
  2. 流式对话：实时响应，提升用户体验
  3. Markdown渲染：支持完整的Markdown语法
  4. 代码高亮：支持多种编程语言的语法高亮

技术栈

  • 后端：Java Spring Boot
  • 前端：原生JavaScript + CSS3
  • Markdown：marked.js
  • 代码高亮：Prism.js
